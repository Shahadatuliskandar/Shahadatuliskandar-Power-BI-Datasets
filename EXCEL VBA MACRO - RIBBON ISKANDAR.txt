Sub Isk_RandNRIC()
Dim yrStr As String
Dim mtStr As String
Dim dyStr As String
Dim stStr As String
Dim idStr As String
For Each cell In Selection
    yrStr = Right(WorksheetFunction.RandBetween(1957, 2020), 2)
    mtStr = WorksheetFunction.Text(WorksheetFunction.RandBetween(1, 12), "00")
    dyStr = WorksheetFunction.Text(WorksheetFunction.RandBetween(1, 28), "00")
    stStr = WorksheetFunction.Text(WorksheetFunction.RandBetween(1, 14), "00")
    idStr = WorksheetFunction.Text(WorksheetFunction.RandBetween(1, 9999), "0000")
    cell.Value = "'" & yrStr & mtStr & dyStr & stStr & idStr
Next cell
Columns.AutoFit
End Sub

Sub Isk_ExtractDOB()
Dim columnCount As Integer
Dim startCell As Range
Dim endCell As Range
columnCount = ActiveSheet.UsedRange.Columns.Count
Set startCell = ActiveCell.End(xlUp).Offset(1, 0)
Set endCell = startCell.End(xlDown)
Cells(startCell.Row - 1, columnCount + 1).Value = "DOB"
Dim yrNRIC As String
Dim mtNRIC As String
Dim dyNRIC As String
For Each cell In Range(startCell, endCell)
    yrNRIC = WorksheetFunction.Text(Left(cell.Value, 2), "00")
    mtNRIC = WorksheetFunction.Text(Mid(cell.Value, 3, 2), "00")
    dyNRIC = WorksheetFunction.Text(Mid(cell.Value, 5, 2), "00")
    Cells(cell.Row, columnCount + 1).Value = DateValue(dyNRIC & "-" & mtNRIC & "-" & yrNRIC)
Next cell
Columns.AutoFit
End Sub

Sub Isk_ReportPages()
Application.ScreenUpdating = False
Dim startCell As Range
Dim endCell As Range
Dim dataRange As Range
Dim activecellColumn As Integer
Dim regionStartColumn As Integer
Dim arr1() As Variant
Dim ws As Worksheet
Set startCell = ActiveCell.End(xlUp).Offset(1, 0)
Set endCell = startCell.End(xlDown)
Set dataRange = ActiveCell.CurrentRegion
activecellColumn = ActiveCell.Column
regionStartColumn = dataRange.Columns.Column
arr1 = WorksheetFunction.Unique(Range(startCell, endCell))
For Each Item In arr1
    Set ws = Sheets.Add(After:=ActiveSheet)
    ws.Name = Item
    dataRange.Copy
    ActiveCell.PasteSpecial (xlPasteAll)
    With Selection.Borders(xlEdgeLeft)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0
        .Weight = xlThin
    End With
    With Selection.Borders(xlEdgeTop)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0
        .Weight = xlThin
    End With
    With Selection.Borders(xlEdgeBottom)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0
        .Weight = xlThin
    End With
    With Selection.Borders(xlEdgeRight)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0
        .Weight = xlThin
    End With
    With Selection.Borders(xlInsideVertical)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0
        .Weight = xlThin
    End With
    With Selection.Borders(xlInsideHorizontal)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0
        .Weight = xlThin
    End With
    Selection.AutoFilter
    Selection.AutoFilter Field:=activecellColumn - regionStartColumn + 1, Criteria1:=Item
    Columns.AutoFit
    Selection.End(xlUp).End(xlToLeft).Select
Next Item
Application.ScreenUpdating = True
End Sub

Sub Isk_TOC()
Dim flag1 As Boolean
Dim flag2 As Integer
Dim ws As Worksheet
Dim i As Integer
flag1 = False
For Each ws In Worksheets
    If ws.Name = "TOC" Then
'        Debug.Print "TOC existed"
        flag1 = True
        Exit For
    End If
Next ws
If flag1 Then
    flag2 = MsgBox("TOC existed. Do you want to re-create TOC?", vbYesNo)
    If flag2 = vbYes Then
        Sheets("TOC").UsedRange.Clear
        With Sheets("TOC").Cells(1, 1)
            .Value = "Table of Contents"
            .Font.Bold = True
            .Font.Color = vbBlue
        End With
        For i = 2 To Worksheets.Count
            Sheets("TOC").Cells(i, 1).Value = Worksheets(i).Name
            Sheets("TOC").Cells(i, 1).Hyperlinks.Add _
                Anchor:=Sheets("TOC").Cells(i, 1), _
                Address:="", _
                SubAddress:="'" & Worksheets(i).Name & "'!A1"
        Next i
    Else
        Exit Sub
    End If
Else
    Set ws = Sheets.Add(Before:=Sheets(1))
    ws.Name = "TOC"
    With Sheets("TOC").Cells(1, 1)
        .Value = "Table of Contents"
        .Font.Bold = True
        .Font.Color = vbBlue
    End With
    For i = 2 To Worksheets.Count
        Sheets("TOC").Cells(i, 1).Value = Worksheets(i).Name
        Sheets("TOC").Cells(i, 1).Hyperlinks.Add _
            Anchor:=Sheets("TOC").Cells(i, 1), _
            Address:="", _
            SubAddress:="'" & Worksheets(i).Name & "'!A1"
    Next i
End If
End Sub

Sub Isk_ColumnChart()
On Error Resume Next
Dim myChart As ChartObject
If ActiveSheet.ChartObjects.Count < 1 Then Exit Sub
For Each myChart In ActiveSheet.ChartObjects
    With myChart
        If .Chart.ChartType = 51 Then
            .Chart.ShowAllFieldButtons = False
            .Chart.Legend.Delete
            .Chart.SetElement (msoElementDataLabelInsideBase)
            .Chart.ApplyDataLabels
            .Width = WorksheetFunction.MRound(300, 15)
            .Height = WorksheetFunction.MRound(300, 14.4)
            .Placement = xlFreeFloating
            .Chart.ChartGroups(1).VaryByCategories = True
            With .Chart.FullSeriesCollection(1)
                .DataLabels.Orientation = xlUpward
                .DataLabels.Format.TextFrame2.Orientation = msoTextOrientationUpward
                .DataLabels.Format.TextFrame2.TextRange.Font.Fill.ForeColor.RGB = RGB(255, 255, 255)
                .DataLabels.Format.TextFrame2.TextRange.Font.Bold = True
                .Format.Shadow.Type = msoShadow21
            End With
            With .Chart.Axes(xlValue)
                .ScaleType = xlLogarithmic
                .MajorGridlines.Delete
                .Delete
            End With
            With .Chart.ChartArea.Format.Shadow
                .Blur = 10
                .OffsetX = 5
                .OffsetY = 5
                .Transparency = 0.65
            End With
        End If
    End With
Next myChart
End Sub
